## 基础示例

### 1. Hello World - 摆动天线

```python
from reachy_mini import ReachyMini

with ReachyMini() as mini:
    print("已连接到 Reachy Mini! 🤖")
    
    # 摆动天线
    mini.goto_target(antennas=[0.5, -0.5], duration=0.5)
    mini.goto_target(antennas=[-0.5, 0.5], duration=0.5)
    mini.goto_target(antennas=[0, 0], duration=0.5)
    
    print("完成!")
```

### 6. 播放音频

```python
from reachy_mini import ReachyMini
import numpy as np
import time

with ReachyMini(media_backend="default") as mini:
    # 初始化
    mini.media.start_playing()
    
    # 生成正弦波 (440 Hz)
    sample_rate = 16000
    duration = 2.0
    t = np.linspace(0, duration, int(sample_rate * duration))
    audio = np.sin(2 * np.pi * 440 * t)
    audio = np.column_stack([audio, audio])  # 立体声
    
    # 播放
    mini.media.push_audio_sample(audio.astype(np.float32))
    time.sleep(duration)
    
    # 停止
    mini.media.stop_playing()
```

### 10. 实时控制 (set_target)

```python
from reachy_mini import ReachyMini
from reachy_mini.utils import create_head_pose
import numpy as np
import time

with ReachyMini() as mini:
    t0 = time.time()
    
    while time.time() - t0 < 5.0:
        t = time.time() - t0
        
        # 正弦运动
        yaw = 30 * np.sin(2 * np.pi * 0.5 * t)
        pitch = 15 * np.sin(2 * np.pi * 0.3 * t)
        
        head = create_head_pose(yaw=yaw, pitch=pitch, degrees=True)
        antennas = [0.3 * np.sin(2 * np.pi * t), -0.3 * np.sin(2 * np.pi * t)]
        
        mini.set_target(head=head, antennas=antennas)
        time.sleep(0.02)  # 50 Hz
```

### 11. 录制和回放动作

```python
from reachy_mini import ReachyMini
import time

with ReachyMini() as mini:
    # 开始录制
    mini.start_recording()
    
    # 手动移动机器人或发送命令
    mini.goto_target(body_yaw=0.5, duration=1.0)
    mini.goto_target(body_yaw=-0.5, duration=1.0)
    mini.goto_target(body_yaw=0, duration=1.0)
    
    # 停止录制
    recorded = mini.stop_recording()
    
    # 回放
    time.sleep(1)
    mini.play_move(recorded)
```

### 17. WebSocket 实时状态

```python
import asyncio
import websockets
import json

async def listen():
    uri = "ws://localhost:8000/api/state/ws/full"
    async with websockets.connect(uri) as websocket:
        while True:
            data = await websocket.recv()
            state = json.loads(data)
            print(f"头部位置: {state.get('head', {})}")

asyncio.run(listen())
```

### 18. 应用模板

```python
import threading
import time
import numpy as np

from reachy_mini import ReachyMini, ReachyMiniApp
from reachy_mini.utils import create_head_pose

class MyCustomApp(ReachyMiniApp):
    custom_app_url: str | None = "http://0.0.0.0:8042"
    
    def run(self, reachy_mini: ReachyMini, stop_event: threading.Event):
        t0 = time.time()
        
        while not stop_event.is_set():
            t = time.time() - t0
            
            # 复杂运动模式
            yaw = 30.0 * np.sin(2.0 * np.pi * 0.2 * t)
            pitch = 15.0 * np.sin(2.0 * np.pi * 0.3 * t)
            roll = 10.0 * np.sin(2.0 * np.pi * 0.1 * t)
            
            head_pose = create_head_pose(
                yaw=yaw, pitch=pitch, roll=roll, degrees=True
            )
            
            # 天线运动
            a = np.deg2rad(25.0 * np.sin(2.0 * np.pi * 0.5 * t))
            antennas = np.array([a, -a])
            
            # 身体旋转
            body_yaw = np.deg2rad(20.0 * np.sin(2.0 * np.pi * 0.15 * t))
            
            reachy_mini.set_target(
                head=head_pose,
                antennas=antennas,
                body_yaw=body_yaw
            )
            time.sleep(0.02)

if __name__ == "__main__":
    app = MyCustomApp()
    try:
        app.wrapped_run()
    except KeyboardInterrupt:
        app.stop()
```

### 19. 检查连接状态

```python
from reachy_mini import ReachyMini

try:
    with ReachyMini() as mini:
        print("连接成功!")
        
        # 检查 daemon 状态
        status = mini.client.get_status()
        print(f"Daemon 状态: {status}")
        
        # 检查电机
        print(f"电机状态: {mini.state}")
        
except ConnectionError as e:
    print(f"连接失败: {e}")
    print("请确保 daemon 正在运行")
except Exception as e:
    print(f"错误: {e}")
```
