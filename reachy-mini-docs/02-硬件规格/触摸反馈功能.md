# 触摸反馈功能详解

## 概述

Reachy Mini **没有专用的触摸传感器**，但可以通过多种方式实现触摸反馈功能。

## 传感器支持情况

| 传感器类型 | Reachy Mini | 说明 |
|------------|-------------|------|
| **触摸传感器** | ❌ | 没有电容/压力触摸 |
| **力传感器** | ❌ | 没有力/扭矩传感器 |
| **电流检测** | ✅ | 可以检测电机电流 |
| **位置反馈** | ✅ | 精确的位置编码器 |
| **IMU** | ⚠️ | 仅 Wireless 版本 |

## 实现触摸反馈的方式

### 方案 1: 电流检测 (推荐)

Dynamixel 电机支持**电流检测**，可以间接感知外力。

**原理:**
```
当机器人被推动时，电机会产生反向电流
通过检测电流变化，可以感知外力

电流 → 力 的关系:
- 电流增大 = 电机在抵抗外力
- 电流为 0 = 无外力
- 电流反向 = 被外力推动
```

**实现代码:**

```python
import numpy as np

class ForceDetector:
    def __init__(self, threshold=50):  # mA
        self.threshold = threshold
        self.baseline_current = None
    
    def calibrate(self, current_readings):
        """校准基线电流"""
        self.baseline_current = np.mean(current_readings)
    
    def detect_force(self, current):
        """检测外力"""
        if self.baseline_current is None:
            return False
        
        delta = abs(current - self.baseline_current)
        return delta > self.threshold
```

### 方案 2: 合规模式 (Compliance Mode)

启用重力补偿模式，让机器人变得"柔软"，可以被手动移动。

```python
with ReachyMini() as mini:
    # 启用重力补偿模式
    mini.enable_gravity_compensation()
    
    print("机器人现在是合规的，可以被推动")
    print("按下 Ctrl+C 退出")
    
    try:
        while True:
            # 读取电机状态（包括电流）
            state = mini.state
            
            # 检测是否被推动
            print(f"电机状态: {state}")
            time.sleep(0.1)
    except KeyboardInterrupt:
        pass
    
    # 恢复正常模式
    mini.enable_motors()
```

**电机模式说明:**

| 模式 | 函数 | 说明 |
|------|------|------|
| **僵硬模式** | `enable_motors()` | 保持位置，抵抗外力 |
| **松弛模式** | `disable_motors()` | 无动力，可自由移动 |
| **合规模式** | `enable_gravity_compensation()` | 补偿重力，可被推动 |

### 方案 3: 位置误差检测

通过目标位置和实际位置的差异来检测碰撞。

```python
import numpy as np

def detect_collision(mini, threshold=0.1):
    """
    通过位置误差检测碰撞
    
    原理：如果目标位置和实际位置差异大，
    说明有外力阻止机器人到达目标
    """
    # 获取目标位置
    target = mini.get_target_pose()
    
    # 获取实际位置
    actual = mini.get_current_head_pose()
    
    # 计算误差
    error = np.linalg.norm(target[:3, 3] - actual[:3, 3])
    
    return error > threshold

# 使用
with ReachyMini() as mini:
    mini.goto_target(z=10, mm=True, duration=2.0)
    
    while True:
        if detect_collision(mini):
            print("碰撞检测！")
            mini.disable_motors()  # 安全停止
            break
        time.sleep(0.02)
```

### 方案 4: 天线作为触摸按钮 (最实用)

天线可以作为**物理按钮**使用，这是最简单实用的触摸反馈方式。

```python
import time
from reachy_mini import ReachyMini

def detect_antenna_press(mini, antenna_id, threshold=0.3):
    """
    检测天线是否被按下
    
    原理：天线被推动时，位置会发生变化
    
    参数:
        antenna_id: 0=左天线, 1=右天线
        threshold: 触发阈值 (弧度)
    """
    _, antennas = mini.get_current_joint_positions()
    
    if antenna_id == 0:
        return abs(antennas[0]) > threshold
    else:
        return abs(antennas[1]) > threshold

# 使用
with ReachyMini() as mini:
    # 启用重力补偿，让天线可以被推动
    mini.enable_gravity_compensation()
    
    print("推动天线作为按钮...")
    print("左天线 = 按钮 1")
    print("右天线 = 按钮 2")
    print("Ctrl+C 退出")
    
    try:
        while True:
            if detect_antenna_press(mini, 0):
                print("左天线被按下！")
                # 做出反应
                mini.goto_target(antennas=[0.5, -0.5], duration=0.3)
                mini.goto_target(antennas=[0, 0], duration=0.3)
                time.sleep(0.5)
            
            if detect_antenna_press(mini, 1):
                print("右天线被按下！")
                mini.goto_target(antennas=[-0.5, 0.5], duration=0.3)
                mini.goto_target(antennas=[0, 0], duration=0.3)
                time.sleep(0.5)
            
            time.sleep(0.02)
    except KeyboardInterrupt:
        pass
```

## 完整应用示例

### 示例 1: 天线按钮游戏

```python
import time
import random
from reachy_mini import ReachyMini
from reachy_mini.motion.recorded_move import RecordedMoves

class AntennaButtonGame:
    def __init__(self, mini):
        self.mini = mini
        self.moves = RecordedMoves("pollen-robotics/reachy-mini-emotions-library")
        self.mini.enable_gravity_compensation()
        self.score = 0
    
    def wait_for_press(self, timeout=5.0):
        """等待天线按下"""
        start_time = time.time()
        
        while time.time() - start_time < timeout:
            _, antennas = self.mini.get_current_joint_positions()
            
            if abs(antennas[0]) > 0.5:
                return 'left'
            if abs(antennas[1]) > 0.5:
                return 'right'
            
            time.sleep(0.02)
        
        return None
    
    def play_round(self):
        """玩一轮"""
        # 随机选择左右
        target = random.choice(['left', 'right'])
        
        # 提示
        if target == 'left':
            print("按左天线！")
            self.mini.goto_target(antennas=[0.8, 0], duration=0.3)
        else:
            print("按右天线！")
            self.mini.goto_target(antennas=[0, -0.8], duration=0.3)
        
        # 等待输入
        result = self.wait_for_press(timeout=3.0)
        
        # 判断结果
        if result == target:
            print("正确！")
            self.score += 1
            self.mini.play_move(self.moves.get("happy"), initial_goto_duration=0.5)
        else:
            print("错误！")
            self.mini.play_move(self.moves.get("sad"), initial_goto_duration=0.5)
        
        # 回到中间
        self.mini.goto_target(antennas=[0, 0], duration=0.3)
    
    def play(self, rounds=5):
        """玩多轮"""
        print(f"开始游戏！共 {rounds} 轮")
        
        for i in range(rounds):
            print(f"\n第 {i+1} 轮:")
            self.play_round()
            time.sleep(1)
        
        print(f"\n游戏结束！得分: {self.score}/{rounds}")

# 使用
with ReachyMini() as mini:
    game = AntennaButtonGame(mini)
    game.play(rounds=5)
```

### 示例 2: 触摸触发反应

```python
import time
from reachy_mini import ReachyMini
from reachy_mini.motion.recorded_move import RecordedMoves

class TouchReactive:
    def __init__(self, mini):
        self.mini = mini
        self.moves = RecordedMoves("pollen-robotics/reachy-mini-emotions-library")
        self.mini.enable_gravity_compensation()
        self.last_touch_time = 0
        self.cooldown = 2.0  # 秒
    
    def check_touch(self):
        """检查触摸"""
        current_time = time.time()
        
        # 冷却时间
        if current_time - self.last_touch_time < self.cooldown:
            return None
        
        _, antennas = self.mini.get_current_joint_positions()
        
        # 检查左天线
        if antennas[0] > 0.5:
            self.last_touch_time = current_time
            return 'left'
        
        # 检查右天线
        if antennas[1] < -0.5:
            self.last_touch_time = current_time
            return 'right'
        
        # 检查头部被推动 (通过位置变化)
        head_pose = self.mini.get_current_head_pose()
        if abs(head_pose[0, 3]) > 0.02 or abs(head_pose[1, 3]) > 0.02:
            self.last_touch_time = current_time
            return 'head'
        
        return None
    
    def react(self, touch_type):
        """对触摸做出反应"""
        if touch_type == 'left':
            print("左天线被触摸 - 开心！")
            self.mini.play_move(self.moves.get("happy"), initial_goto_duration=0.5)
        
        elif touch_type == 'right':
            print("右天线被触摸 - 惊讶！")
            self.mini.play_move(self.moves.get("surprised"), initial_goto_duration=0.5)
        
        elif touch_type == 'head':
            print("头部被触摸 - 害羞！")
            self.mini.play_move(self.moves.get("shy"), initial_goto_duration=0.5)
    
    def run(self):
        """运行主循环"""
        print("触摸机器人来互动...")
        print("左天线 = 开心")
        print("右天线 = 惊讶")
        print("头部 = 害羞")
        print("Ctrl+C 退出")
        
        try:
            while True:
                touch = self.check_touch()
                if touch:
                    self.react(touch)
                time.sleep(0.02)
        except KeyboardInterrupt:
            pass

# 使用
with ReachyMini() as mini:
    reactive = TouchReactive(mini)
    reactive.run()
```

### 示例 3: 碰撞检测安全系统

```python
import time
import numpy as np
from reachy_mini import ReachyMini

class CollisionSafety:
    def __init__(self, mini, position_threshold=0.05, imu_threshold=15.0):
        self.mini = mini
        self.position_threshold = position_threshold  # 米
        self.imu_threshold = imu_threshold  # m/s²
        self.is_emergency = False
    
    def check_position_error(self):
        """检查位置误差"""
        target = self.mini.get_target_pose()
        actual = self.mini.get_current_head_pose()
        
        error = np.linalg.norm(target[:3, 3] - actual[:3, 3])
        return error > self.position_threshold
    
    def check_imu_collision(self):
        """检查 IMU 碰撞检测"""
        try:
            accel = self.mini.imu["accelerometer"]
            magnitude = np.linalg.norm(accel)
            return magnitude > self.imu_threshold
        except:
            return False
    
    def emergency_stop(self):
        """紧急停止"""
        print("紧急停止！")
        self.is_emergency = True
        self.mini.disable_motors()
    
    def reset(self):
        """重置系统"""
        print("重置系统")
        self.is_emergency = False
        self.mini.enable_motors()
    
    def monitor(self):
        """监控循环"""
        print("碰撞检测安全系统已启动")
        print("Ctrl+C 退出")
        
        try:
            while True:
                if self.is_emergency:
                    time.sleep(0.1)
                    continue
                
                # 检查位置误差
                if self.check_position_error():
                    print("警告: 位置误差过大！")
                    self.emergency_stop()
                    continue
                
                # 检查 IMU 碰撞
                if self.check_imu_collision():
                    print("警告: 检测到碰撞！")
                    self.emergency_stop()
                    continue
                
                time.sleep(0.02)
        except KeyboardInterrupt:
            pass

# 使用
with ReachyMini() as mini:
    safety = CollisionSafety(mini)
    safety.monitor()
```

### 方案 2: 压力传感器 (FSR)

```
Raspberry Pi 4 + FSR 压力传感器 (~$3)
```

```python
import RPi.GPIO as GPIO
import spidev

# 使用 MCP3008 ADC 读取模拟信号
spi = spidev.SpiDev()
spi.open(0, 0)
spi.max_speed_hz = 1000000

def read_adc(channel):
    """读取 ADC 值"""
    cmd = [1, (8 + channel) << 4, 0]
    reply = spi.xfer2(cmd)
    value = ((reply[1] & 3) << 8) + reply[2]
    return value

def read_fsr():
    """读取 FSR 压力值"""
    adc_value = read_adc(0)  # 通道 0
    # 转换为压力值
    # 值越大 = 压力越大
    return adc_value

# 使用
while True:
    pressure = read_fsr()
    if pressure > 100:
        print(f"检测到压力: {pressure}")
    time.sleep(0.1)
```
